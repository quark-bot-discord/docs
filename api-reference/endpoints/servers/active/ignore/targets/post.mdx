---
title: 'Actively Ignore Targets'
api: 'POST /servers/{guildId}/active/ignore/targets'
---

<ParamField path="guild ID" type="string">
  The ID of the guild.
</ParamField>

<ParamField body="id" type="string" required>
  The ID of the target to ignore. (for invites this is the code)
</ParamField>

<ParamField body="type" type="integer" required>
  The type of target to ignore. MEMBER: 0, CHANNEL: 1, EMOJI: 2, GUILD: 3, INVITE: 4, NONE: 5, MESSAGE: 6, ROLE: 7, THREAD: 8, USER: 9
</ParamField>

<ParamField body="logType" type="integer">
  The type ID of the log. See the [log types](/logs/types) for more information.
</ParamField>

<ParamField body="count" type="integer">
  The number of events to ignore.
</ParamField>

<ParamField body="time" type="integer">
  The period to apply the ignore for.
</ParamField>

<ResponseField name="token" type="string" required>
  The `Bearer` token to use for authenticating requests to the Quark API.
</ResponseField>
