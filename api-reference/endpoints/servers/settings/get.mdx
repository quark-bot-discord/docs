---
title: 'Get Settings Configuration'
api: 'GET /servers/{guildId}/settings'
---

<ParamField path="guild ID" type="string">
  The ID of the guild.
</ParamField>

<ParamField query="bot" type="string" default="main">
  The bot to get the configuration for. Either "main" or "pro".
</ParamField>

<ResponseField name="id" type="string" required>
  The ID of the guild.
</ResponseField>

<ResponseField name="options" type="object" required>
  The configuration options for the settings.
</ResponseField>
